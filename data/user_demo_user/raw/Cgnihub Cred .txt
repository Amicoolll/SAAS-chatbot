On EC2 (deploy)
# 1) SSH in
ssh -i ~/.ssh/cognihub-ec2-mumbai.pem ec2-user@3.111.78.19
# 2) Pull latest code
cd ~/backend
BRANCH=$(git branch --show-current)
GIT_SSH_COMMAND='ssh -i ~/.ssh/github_deploy -o IdentitiesOnly=yes' git fetch origin
GIT_SSH_COMMAND='ssh -i ~/.ssh/github_deploy -o IdentitiesOnly=yes' git pull --rebase origin "$BRANCH"
# 3) Install/upgrade dependencies (your requirements already have Python-conditional pins)
source venv/bin/activate
pip install --no-cache-dir -r requirements.txt
# 4) Apply Alembic migrations to the prod DB
export PYTHONPATH=$PWD
set -a; [ -f .env ] && . ./.env; set +a
python -m alembic upgrade head
# 5) Restart the API
# If you created the systemd service (recommended):
sudo systemctl restart cognihub-api
sudo systemctl status cognihub-api --no-pager -l
# If you don't have the service yet, use uvicorn directly:
# (free the port if a root-owned process is holding it)
PID=$(sudo lsof -t -iTCP:8000 -sTCP:LISTEN 2>/dev/null); [ -n "$PID" ] && sudo kill -9 $PID || true
nohup uvicorn app.main:app --host 0.0.0.0 --port 8000 --proxy-headers >/tmp/backend.out 2>&1 &
sleep 2 && tail -n 80 /tmp/backend.out
# 6) Quick health checks
curl -I http://127.0.0.1:8000/docs
curl -I https://api.cognihubai.com/docs
That’s it ✅
